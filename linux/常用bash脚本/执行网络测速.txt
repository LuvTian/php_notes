#!/bin/sh
if [ -f "/usr/bin/yum" ]; then
    RUM="yum"
elif [ -f "/usr/bin/apt-get" ]; then
    RUM="apt-get"
elif [ -f "/usr/bin/dnf" ]; then
    RUM="dnf"
fi;
if [ $RUM == "apt-get" ]; then
    $RUM update -y
else
    $RUM update --exclude=kernel* -y
    $RUM install epel* -y
fi;
$RUM install -y python-pip
easy_install pip
pip install speedtest-cli
speedtest-cli > /root/speedtest.log


#执行网络测速 (结果将保存在/root/speedtest.log).txt
#测试由speedtest.net提供
#执行速度由系统包源决定
